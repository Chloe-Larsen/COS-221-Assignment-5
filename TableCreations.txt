CREATE TABLE Traveller (
userId INT AUTO_INCREMENT PRIMARY KEY,
idNumber INT NOT NULL, 
firstName VARCHAR(64) NOT NULL,
surname VARCHAR(64) NOT NULL,
phoneNumber CHAR(10),
email VARCHAR(254) NOT NULL,
buildingNumber INT NOT NULL,
street VARCHAR(64) NOT NULL,
suburb VARCHAR(64) NOT NULL,
postalCode VARCHAR(6),
dateJoined DATE
);

CREATE TABLE Agencies (
userId INT PRIMARY KEY,
companyName VARCHAR(100) NOT NULL,
firstName VARCHAR(64) NOT NULL,
surname VARCHAR(64) NOT NULL,
phoneNumber CHAR(10),
email VARCHAR(254) NOT NULL,
password VARCHAR(20) NOT NULL,
dateJoined DATE,

CONSTRAINT fk_user FOREIGN KEY (userId) REFERENCES Traveller(userId)
);

CREATE TABLE Destination (
destinationID INT AUTO_INCREMENT PRIMARY KEY,
cityName VARCHAR(64) NOT NULL,
province VARCHAR(64) NOT NULL
);

CREATE TABLE Flight (
flightId INT AUTO_INCREMENT PRIMARY KEY,
airline VARCHAR(64) NOT NULL,
departureDate DATE,
departureTime TIME,
arrivalTime TIME,
);

CREATE TABLE Accommodation (
accommodationId INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(100) NOT NULL,
type ENUM('Hotel', 'Motel', 'B&B', 'Resort'),
buildingNumber INT NOT NULL,
street VARCHAR(64) NOT NULL,
suburb VARCHAR(64) NOT NULL,
postalCode VARCHAR(6) NOT NULL
);

CREATE TABLE Restaurant (
restaurantId INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(100) NOT NULL,
cuisineType TEXT,
buildingNumber INT NOT NULL,
street VARCHAR(64) NOT NULL,
suburb VARCHAR(64) NOT NULL,
postalCode VARCHAR(6) NOT NULL
);

CREATE TABLE Attraction (
attractionId INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(254),
entranceFee DECIMAL(10,2) NOT NULL 
);

CREATE TABLE Review (
reviewId INT AUTO_INCREMENT PRIMARY KEY,
rating INT NOT NULL,
comment TEXT,
timestamp TIMESTAMP
);

CREATE TABLE Package (
packageId INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(254) NOT NULL,
description TEXT,
basePrice DECIMAL(10,2) NOT NULL,
duration INT NOT NULL,
images TEXT, 
averageRating DECIMAL(10,2) NOT NULL
);

CREATE TABLE GroupTrip (
packageId INT PRIMARY KEY,
maxCapacity INT NOT NULL,

CONSTRAINT fk_grouptrip_package FOREIGN KEY (packageId) REFERENCES Package(packageId) ON DELETE CASCADE
);

CREATE TABLE ItineraryDay (
dayNumber INT PRIMARY KEY,
description TEXT
);
 
CREATE TABLE Booking (
bookingId INT AUTO_INCREMENT PRIMARY KEY,
dateBooked DATE,
partySize INT NOT NULL,
totalCost DECIMAL(10,2) NOT NULL,
paymentStatus ENUM('Pending', 'Rejected', 'Approved'),
legNumber INT NOT NULL,
seatClass ENUM('Economy', 'Business', 'Premium Economy', 'First Class')
);
